### 数组按照个数切割
```
function chunkFn(source: string[], index: number) {
  if (!index) return [source]
  index = index * 1 || 1
  const ret: Array<Array<string>> = []
  source.forEach((item, i) => {
    if (i % index === 0) ret.push([])
    ret[ret.length - 1].push(item)
  })
  return ret
}
chunkFn([1,2,3,4,5], 2)
```
